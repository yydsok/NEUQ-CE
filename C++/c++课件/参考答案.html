void sort( int a[], int n )
{
    for(int i=0; i<= n-1; i++)
    {
        int minv=*(a+i);
        int loca = i;
        for (int j=i+1; j<=n-1; j++)
        {
            if (minv > *(a+j))
            {
                minv = *(a+j);
                loca = j;
            }
        }
        *(a+loca) = *(a+i);
        *(a+i) = minv;        
    }
}

输入在一行中给出一个正实数eps。

在一行中按照“sum = S”的格式输出部分和的值S，精确到小数点后六位。题目保证计算结果不超过双精度范围。

4E-2

#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
    double eps,sum=0, item=1.0, dominator = 1.0;
    cin>>eps;
    
    int sign = 1;
    
    if (eps >= 1)
    {
        cout<<fixed<<setprecision(6)<<"sum = "<<item<<endl;
        return 0;
    }
    
    while(fabs(item) > eps)
    {
        item = sign * 1.0 / dominator;
        sum = sum + item;
        dominator += 3;
        sign = - sign;
    }
    
    cout<<"sum = "<<fixed<<setprecision(6)<<sum<<endl;
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;

int prime(int m);
int main()
{
    int m,n, sum = 0;
    cin>>m>>n;
    if (m<0 || n<0) return 0;
    if (m > n)
    {
        int temp = m;
        m = n;
        n = temp;
    }

    for(int i = m+1; i<n;i++)
    {
      if(prime(i))
      {
        sum = sum + i;
      }
    }
    cout<<sum;
    
    return 0;
}

int prime(int m)
{
    if(m<=1)    
        return 0;
    else if(m==2)
        return 1;
    else
    {
        for(int i=2;i<=m/2;i++)
        {
            if(m%i == 0)
            {
                return 0;
            }
        }
        return 1;
    }
}
